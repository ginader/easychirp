  132.         $class_name = get_class($parent);
  133.

  137.  
  138.             foreach ($r->getMethods() as $method)
  139.             {
  140.                 if ($method->isPublic())
  141.                 {
  142.                     $this->methods[] = $method->getName();
  143.                 }
  144.             }
  145.  
  146.             foreach ($r->getProperties() as $prop)
  147.             {
  148.                 if ($prop->isPublic())
  149.                 {
  150.                     $this->properties[] = $prop->getName();
  151.                 }
  152.             }
  153.  
  154.             self::$reflections[$class_name] = array($this->methods, $this->properties);
  155.         }
  156.         else
  157.         {
  158.             list($this->methods, $this->properties) = self::$reflections[$class_name];
  159.         }
  160.     }
